
create table STUD_MARKS(ROLLNO INT, NAME VARCHAR2(20), MARKS INT);
create table RESULT (ROLL_NO INT, NAME VARCHAR2(20), CLASS VARCHAR(20));

insert into STUD_MARKS Values ( 1, 'Ajay', 1100 );
insert into STUD_MARKS Values ( 2, 'Ramesh', 950);
insert into STUD_MARKS Values ( 3, 'Umesh', 850);
insert into STUD_MARKS Values ( 4, 'Suresh', 400);


create or replace PROCEDURE PROC_GRADE AS

R_NO STUD_MARKS.ROLLNO%TYPE;
NAME1 STUD_MARKS.NAME%TYPE;
MRKS STUD_MARKS.MARKS%TYPE;
CURSOR C1 IS SELECT ROLLNO,NAME,MARKS FROM STUD_MARKS;

BEGIN
OPEN C1;
LOOP
FETCH C1 INTO R_NO,NAME1,MRKS ;
EXIT WHEN C1%NOTFOUND;

IF(MRKS<=1500 AND MRKS>=990) THEN
INSERT INTO RESULT VALUES(R_NO,NAME1,'DISTINCTION');
ELSIF(MRKS>=900 AND MRKS<=989)THEN
INSERT INTO RESULT VALUES(R_NO,NAME1,'FIRST CLASS');
ELSIF(MRKS>=825 AND MRKS<=899)THEN
INSERT INTO RESULT VALUES(R_NO,NAME1,'HIGHER 2ND CLASS');
ELSE
INSERT INTO RESULT VALUES(R_NO,NAME1,'FAIL');
END IF;
END LOOP;
CLOSE C1;
END PROC_GRADE;

EXECUTE PROC_GRADE;

SELECT * FROM RESULT;
